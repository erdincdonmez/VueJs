    3.             new Vue({

  3. bir div (veya herhangi bir elemen oluştur, idsini ver
    4.           <div id="app">

  4. yeni uygulamada yöneteceğin elementi seç
    5.           new Vue({
            el : "#app",//id'si app olan div elementini seçtik.
        }
        );
  5. data ları ekle
    6.           data:{
                title : "test başlık",
                buttonTitle : "Tıkla gelsin",

            },
  6. html elementlerine dataları yazdır.
    7.         <div id="app">

            <button>{{buttonTitle}}</button>
            <p>{{aciklama}}</p>
            <p v-text="aciklama"></p>
            <a v-bind:href="link" v-bind:id="idsi">erdincdonmez</a>
            <a :href="link" v-bind:id="idsi">erdincdonmez</a>
            <p v-html="veri1"></p>
