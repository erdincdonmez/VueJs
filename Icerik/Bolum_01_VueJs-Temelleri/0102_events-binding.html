Yukarıdaki örneğin kodları:

    <div id="app">
      <h1>event (olay) örnekleri.</h1>
      <h2>ilk üç kutuya bir sayı yaz ve enter tuşuna bas</h2>
        <input v-on:keypress="ekle($event)" type="text">
        <input @keypress="ekle($event)" type="text">
        <input @keypress.enter="ekle1($event)" type="text">
        <input @input="aktar($event)" type="text">
        <button v-on:keypress="ekle()">Ekle</button>
        <button v-on:click="artir()">Değerleri artırmak için tıkla</button>
        <button v-on:mouseover="artir()">Değerleri artırmak için üzerime gel</button>
        <button v-on:click="bos()">Sabit mesaj göster</button>
        <button v-on:click="bos1('Buton ile gönderilen mesaj')">İstenilen mesajı göster</button>
        <p v-text="sayi">Sayi</p>
        <p>{{sayi}}</p>
        <p>{{mesaj}}</p>
    </div>
    <script>
        new Vue({
            el : "#app",
            data:{
                sayi:0,
                mesaj : "4.Kutuya bir şeyler yaz."
            },
            methods : {
                artir(){
                    this.sayi++;
                },
                bos(){
                    alert("Sabit mesaj.");
                },
                ekle(gelen){
                    if (gelen.keyCode == 13)    
                    this.sayi+= parseInt(gelen.target.value);
                },
                ekle1(gelen){
                    this.sayi+= parseInt(gelen.target.value);
                },
                bos1(pp){
                    alert(pp);
                },
                aktar(gelen){
                    this.mesaj= gelen.target.value;
                },
            }
        });
    </script>
